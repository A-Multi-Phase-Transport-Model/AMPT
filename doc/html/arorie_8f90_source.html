16  If (p1>=p3) Then
17  psi = .5*ulangl(p1**2+p3**2*cos(2.*bet), -p3**2*sin(2.*bet))
18  pt1 = p1*sin(psi)
19  pz1 = p1*cos(psi)
20  pt3 = p3*sin(psi+bet)
21  pz3 = p3*cos(psi+bet)
22  Else If (p3>p1) Then

25  pz1 = -p1*cos(bet+psi)

28  End If

34  p(jl+2, 4) = e3
35  p(jl+2, 1) = pt3*sin(del)
36  p(jl+2, 2) = -pt3*cos(del)
37  p(jl+2, 3) = pz3
38  p(jl+1, 4) = w - e1 - e3
39  p(jl+1, 1) = -p(jl, 1) - p(jl+2, 1)
40  p(jl+1, 2) = -p(jl, 2) - p(jl+2, 2)
41  p(jl+1, 3) = -p(jl, 3) - p(jl+2, 3)
