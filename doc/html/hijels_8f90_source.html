1 Subroutine hijels(psc1, psc2)
2  Implicit Double Precision (d)
3  dimension psc1(5), psc2(5)
4  Common /hparnt/hipr1(100), ihpr2(50), hint1(100), ihnt2(50)
5  Common /rndf77/nseed
6  Save
7  cc = 1.0 - hint1(12)/hint1(13)
8  rr = (1.0-cc)*hint1(13)/hint1(12)/(1.0-hipr1(33)) - 1.0
9  bb = 0.5*(3.0+rr+sqrt(9.0+10.0*rr+rr**2))
10  ep = sqrt((psc1(1)-psc2(1))**2+(psc1(2)-psc2(2))**2+(psc1(3)-psc2(3))**2)
11  If (ep<=0.1) Return
12  els0 = 98.0/ep + 52.0*(1.0+rr)**2
13  pcm1 = psc1(1) + psc2(1)
14  pcm2 = psc1(2) + psc2(2)
15  pcm3 = psc1(3) + psc2(3)
16  ecm = psc1(4) + psc2(4)
17  am1 = psc1(5)**2
18  am2 = psc2(5)**2
19  amm = ecm**2 - pcm1**2 - pcm2**2 - pcm3**2
20  If (amm<=psc1(5)+psc2(5)) Return
21  pmax = (amm**2+am1**2+am2**2-2.0*amm*am1-2.0*amm*am2-2.0*am1*am2)/4.0/amm
22  pmax = abs(pmax)
23  20 tt = ranart(nseed)*min(pmax, 1.5)
24  els = 98.0*exp(-2.8*tt)/ep + 52.0*exp(-9.2*tt)*(1.0+rr*exp(-4.6*(bb-1.0)*tt))**2
25  If (ranart(nseed)>els/els0) Goto 20
26  phi = 2.0*hipr1(40)*ranart(nseed)
27  dbx = dble(pcm1/ecm)
28  dby = dble(pcm2/ecm)
29  dbz = dble(pcm3/ecm)
30  db = dsqrt(dbx**2+dby**2+dbz**2)
31  If (db>0.99999999d0) Then
32  dbx = dbx*(0.99999999d0/db)
33  dby = dby*(0.99999999d0/db)
34  dbz = dbz*(0.99999999d0/db)
35  db = 0.99999999d0
36  Write (6, *) ' (HIJELS) boost vector too large'
37  End If
38  dga = 1d0/sqrt(1d0-db**2)
39  dp1 = dble(sqrt(tt)*sin(phi))
40  dp2 = dble(sqrt(tt)*cos(phi))
41  dp3 = dble(sqrt(pmax-tt))
42  dp4 = dble(sqrt(pmax+am1))
43  dbp = dbx*dp1 + dby*dp2 + dbz*dp3
44  dgabp = dga*(dga*dbp/(1d0+dga)+dp4)
45  psc1(1) = sngl(dp1+dgabp*dbx)
46  psc1(2) = sngl(dp2+dgabp*dby)
47  psc1(3) = sngl(dp3+dgabp*dbz)
48  psc1(4) = sngl(dga*(dp4+dbp))
49  dp1 = -dble(sqrt(tt)*sin(phi))
50  dp2 = -dble(sqrt(tt)*cos(phi))
51  dp3 = -dble(sqrt(pmax-tt))
52  dp4 = dble(sqrt(pmax+am2))
53  dbp = dbx*dp1 + dby*dp2 + dbz*dp3
54  dgabp = dga*(dga*dbp/(1d0+dga)+dp4)
55  psc2(1) = sngl(dp1+dgabp*dbx)
56  psc2(2) = sngl(dp2+dgabp*dby)
57  psc2(3) = sngl(dp3+dgabp*dbz)
58  psc2(4) = sngl(dga*(dp4+dbp))
59  Return
60 End Subroutine hijels
