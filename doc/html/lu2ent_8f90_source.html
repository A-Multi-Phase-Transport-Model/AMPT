1 Subroutine lu2ent(ip, kf1, kf2, pecm)
2  Common /lujets/n, k(9000, 5), p(9000, 5), v(9000, 5)
3  Save /lujets/
4  Common /ludat1/mstu(200), paru(200), mstj(200), parj(200)
5  Save /ludat1/
6  Common /ludat2/kchg(500, 3), pmas(500, 4), parf(2000), vckm(4, 4)
7  Save /ludat2/
8  mstu(28) = 0
9  If (mstu(12)>=1) Call lulist(0)
10  ipa = max(1, iabs(ip))
11  If (ipa>mstu(4)-1) Call luerrm(21, '(LU2ENT:) writing outside LUJETS memory')
12  kc1 = lucomp(kf1)
13  kc2 = lucomp(kf2)
14  If (kc1==0 .Or. kc2==0) Call luerrm(12, '(LU2ENT:) unknown flavour code')
15  pm1 = 0.
16  If (mstu(10)==1) pm1 = p(ipa, 5)
17  If (mstu(10)>=2) pm1 = ulmass(kf1)
18  pm2 = 0.
19  If (mstu(10)==1) pm2 = p(ipa+1, 5)
20  If (mstu(10)>=2) pm2 = ulmass(kf2)
21  Do i = ipa, ipa + 1
22  Do j = 1, 5
23  k(i, j) = 0
24  p(i, j) = 0.
25  v(i, j) = 0.
26  End Do
27  End Do
28  kq1 = kchg(kc1, 2)*isign(1, kf1)
29  kq2 = kchg(kc2, 2)*isign(1, kf2)
30  If (kq1+kq2/=0 .And. kq1+kq2/=4) Call luerrm(2, '(LU2ENT:) unphysical flavour combination')
31  k(ipa, 2) = kf1
32  k(ipa+1, 2) = kf2
33  If (ip>=0) Then
34  k(ipa, 1) = 1
35  If (kq1/=0 .And. kq2/=0) k(ipa, 1) = 2
36  k(ipa+1, 1) = 1
37  Else
38  If (kq1==0 .Or. kq2==0) Call luerrm(2, '(LU2ENT:) requested flavours can not develop parton shower')
39  k(ipa, 1) = 3
40  k(ipa+1, 1) = 3
41  k(ipa, 4) = mstu(5)*(ipa+1)
42  k(ipa, 5) = k(ipa, 4)
43  k(ipa+1, 4) = mstu(5)*ipa
44  k(ipa+1, 5) = k(ipa+1, 4)
45  End If
46  If (pecm<=pm1+pm2) Call luerrm(13, '(LU2ENT:) energy smaller than sum of masses')
47  pa = sqrt(max(0.,(pecm**2-pm1**2-pm2**2)**2-(2.*pm1*pm2)**2))/(2.*pecm)
48  p(ipa, 3) = pa
49  p(ipa, 4) = sqrt(pm1**2+pa**2)
50  p(ipa, 5) = pm1
51  p(ipa+1, 3) = -pa
52  p(ipa+1, 4) = sqrt(pm2**2+pa**2)
53  p(ipa+1, 5) = pm2
54  n = ipa + 1
55  If (ip==0) Call luexec
56  Return
57 End Subroutine lu2ent
