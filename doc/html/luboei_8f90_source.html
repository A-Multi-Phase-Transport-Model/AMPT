44  If (ibe==4 .And. max(nbe(4)-nbe(3),nbe(5)-nbe(4),nbe(6)-nbe(5),nbe(7)-nbe(6))<=1) Goto 180

110  240 End Do

114  Do j = 1, 3
115  p(i, j) = fac*p(i, j)
116  End Do
117  p(i, 4) = sqrt(p(i,5)**2+p(i,1)**2+p(i,2)**2+p(i,3)**2)
118  260 End Do
