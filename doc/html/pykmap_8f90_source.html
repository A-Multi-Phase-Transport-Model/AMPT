31  tau = taure + gamre*tan(alow+(aupp-alow)*vvar)

44  If (iset(isub)>=3) yst = 0.5*log(vint(26))

49  Else

53  End If

55  Else If (ivar==3) Then

69  Else

86  End If

91  rmpmax = max(rm34, rsqm+ctpmax)

129  End If
