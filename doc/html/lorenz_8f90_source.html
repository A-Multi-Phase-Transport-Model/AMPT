19  pynew = -gam*bey*energy + (gam-1.d0)*bex*bey/beta2*px + (1.d0+(gam-1.d0)*bey**2/beta2)*py + (gam-1.d0)*bey*bez/beta2*pz
20  pznew = -gam*bez*energy + (gam-1.d0)*bex*bez/beta2*px + (gam-1.d0)*bey*bez/beta2*py + (1.d0+(gam-1.d0)*bez**2/beta2)*pz
21  End If
